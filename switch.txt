
/****switch****/

#include <stdio.h>
void main() {
    int val;
    printf("enter a value between 1 to 7");
    scanf("%d",&val);
    switch(val){
        case 1:
        printf("monday");
        break;
        case 2:
        printf("tuesday");
        break;
         case 3:
        printf("wednesday");
        break;
         case 4:
        printf("thursday");
        break;
         case 5:
        printf("friday");
        break;
         case 6:
        printf("satuarday");
        break;
         case 7:
        printf("sunday");
        break;
        default:
        printf("invalid");
    }
    

    return 0;
}
