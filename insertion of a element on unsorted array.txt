#include <stdio.h>

int insertSorted(int a[], int n, int key, int capacity)
{

	if (n >= capacity)
		return n;

	a[n] = key;

	return (n + 1);
}


int main()
{
	int a[20] = { 67, 98, 12, 10, 51, 23 };
	int capacity = sizeof(a) / sizeof(a[0]);
	int n = 6;
	int i, key = 30;

	printf("\n Before Insertion: ");
	for (i = 0; i < n; i++)
		printf("%d ", a[i]);

	n = insertSorted(a, n, key, capacity);

	printf("\n After Insertion: ");
	for (i = 0; i < n; i++)
		printf("%d ", a[i]);

	return 0;
}
